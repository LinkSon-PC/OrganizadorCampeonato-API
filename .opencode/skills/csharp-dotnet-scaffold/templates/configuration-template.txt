// ============================================================================
// CONFIGURATION RULES
// ============================================================================
//
// SCALAR PROPERTIES:
// - String required: builder.Property(x => x.Name).IsRequired().HasMaxLength(150);
// - String optional: builder.Property(x => x.Name).HasMaxLength(150);
// - Enum: no config needed (stored as int)
// - DateTime: no config needed (stored as datetime2)
// - Guid FK: no config needed (stored as uniqueidentifier)
//
// RELATIONSHIPS:
// - One-to-Many: builder.HasMany(x => x.Collection).WithOne(x => x.Parent).HasForeignKey(x => x.ParentId).OnDelete(DeleteBehavior.NoAction);
// - Many-to-One: builder.HasOne(x => x.Parent).WithMany(x => x.Collection).HasForeignKey(x => x.ParentId).OnDelete(DeleteBehavior.NoAction);
// - One-to-One: builder.HasOne(x => x.Related).WithOne(x => x.Parent).HasForeignKey<Related>(x => x.Id).OnDelete(DeleteBehavior.NoAction);
// - Optional FK: add .IsRequired(false) before .OnDelete()
//
// INDEXES:
// - Unique: builder.HasIndex(x => x.Property).IsUnique();
// - Non-unique: builder.HasIndex(x => x.Property);
//
// ALL RELATIONSHIPS MUST USE DeleteBehavior.NoAction
